#define LL long long

//为自己加油O(∩_∩)O~

const long long  mod =1000000007;  //这个必须是质数

LL num[1000000],op[1000000];
LL cul(int n,int o)
{
    if (o==0) num[n-1]=num[n-1]+num[n];
    if (o==1) num[n-1]=num[n-1]-num[n];
    if (o==2) num[n-1]=num[n-1]*num[n];
    if (o==3) num[n-1]=num[n-1]/num[n];
}
LL suan(string z)
{
    int n=0,p=0;
    int len=z.size();
    LL x=0;
    LL hei=0;
    for (int j=0;j<len;j++){
        if (z[j]>='0' && z[j]<='9'){
            x=x*10+(z[j]-'0');
        }else if (z[j]!=' '){
            if (x!=0){
                num[n++]=x;
                x=0;
            }
            if (z[j]=='+') hei=0;
            if (z[j]=='-') hei=1;
            if (z[j]=='*') hei=2;
            if (z[j]=='/') hei=3;
            if (z[j]=='(') hei=4;
            if (z[j]==')') hei=5;
            if (hei==4){
                op[p++]=hei;
            }else if (hei==5){
                while (p>0 && op[p-1]!=4){
                    cul(--n,op[--p]);
                }
            }else {
                while (p>0 && op[p-1]<=3 && op[p-1]/2>=hei/2){
                    cul(--n,op[--p]);
                }
                op[p++]=hei;
            }
        }
    }
    if (x!=0){
        num[n++]=x;
        x=0;
    }
    while (p>0) cul(--n,op[--p]);
    return num[0];
}
int main()
{
    int T;
    scanf("%d",&T);
    while (T--){
        string s;
        cin>>s;
        printf("%lld\n",suan(s));
    }
    return 0;
}
