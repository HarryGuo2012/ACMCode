//静态链表（前向星）
int first[maxn];//maxn为点数，first[i]表示从i出发第一条边保存的位置。
int tot; //当前边数
struct Edge
{
    int u,v,next;//u:起点  v:终点  next:下条边的位置
    int w1,w2,...wn  边上的其他附加信息
}edge[maxm]; //maxm为边数，edge保存所有边


memset(first,-1,sizeof(first));tot=0;

void addedge(int u,int v,int w1....wn)
{
    edge[tot].u=u;
    edge[tot].v=v;
    edge[tot].w1...wn=w1....wn;
    edge[tot].next=first[u];
    first[u]=tot++;
}
