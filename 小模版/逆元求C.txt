//吴神美如画
#define LL long long

const long long  mod =1000000007;  //这个必须是质数

LL jiecheng[1048579];
LL extend_gcd(LL a,LL b,LL &x,LL &y)
{
    if(b==0)
    {
        x=1;y=0;return a;
    }
    LL ans=extend_gcd(b,a%b,x,y);
    LL tmp=x;
    x=y;
    y=tmp-(a/b)*y;
    return ans;
}
LL inverse(LL b)
{
    LL x,y;
    extend_gcd(b,mod,x,y);
    return (x%mod+mod)%mod;
}
LL comb(LL n,LL m)
{
    LL t1=1,t2=1;
    if(m==0)return 1;
    if(m<n/2)m=n-m;
    t1=jiecheng[n];
    t2=jiecheng[n-m]*jiecheng[m]%mod;
    return t1*inverse(t2)%mod;
}

LL Lucas(LL n,LL m)
{
    if (n<mod) return comb(n,m)%mod;
    return Lucas(n%mod,m%mod)*Lucas(n/mod,m/mod)%mod;
}
